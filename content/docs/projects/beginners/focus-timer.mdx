# 构建专注时钟应用

在本教程中，我们将创建一个专注时钟（也称为番茄钟）Web应用，它可以帮助你使用时间管理技术来提高工作效率。这个应用将包含计时器功能、工作和休息周期的交替，以及通知提醒。

```
![专注时钟应用截图](/images/docs/projects/focus-timer-screenshot.png)
```

## 项目概述

**预计完成时间**: 3小时  
**难度等级**: ⭐⭐☆☆☆ (入门级+)  
**前置知识**: 基础HTML、CSS和JavaScript  
**项目目标**: 创建一个专注时钟Web应用，帮助用户管理工作和休息时间

### 你将学到什么

- JavaScript计时器和时间操作
- 浏览器通知API的使用
- 音频元素的控制
- 使用LocalStorage保存用户设置
- CSS动画和过渡效果
- 响应式设计技巧

### 完成项目后的成果

一个功能完整的专注时钟应用，包括：
- 可调整的工作和休息时间设置
- 视觉和音频提醒
- 会话计数和统计
- 暂停/继续功能
- 自动在工作和休息模式之间切换
- 响应式界面，适合所有设备

## 开发环境准备

这个项目需要以下工具：

1. 代码编辑器（推荐Visual Studio Code或Cursor）
2. 现代浏览器（Chrome、Firefox或Edge）
3. 几个简单的音频文件（可从免费音效库获取）

我们将创建以下文件结构：

## 步骤3：实现JavaScript功能

现在创建`script.js`文件，实现应用的核心逻辑：

```javascript
// DOM元素
const minutesElement = document.getElementById('minutes');
const secondsElement = document.getElementById('seconds');
const modeIndicatorElement = document.getElementById('mode-indicator');
const timerElement = document.getElementById('timer');
const startButton = document.getElementById('start-btn');
const pauseButton = document.getElementById('pause-btn');
const resetButton = document.getElementById('reset-btn');
const sessionsCountElement = document.getElementById('sessions-count');
const sessionsGoalElement = document.getElementById('sessions-goal');
const settingsToggleButton = document.getElementById('settings-toggle');
const settingsContainer = document.getElementById('settings-container');
const saveSettingsButton = document.getElementById('save-settings');
const bellSound = document.getElementById('bell-sound');
const tickSound = document.getElementById('tick-sound');

// 设置输入元素
const workDurationInput = document.getElementById('work-duration');
const shortBreakDurationInput = document.getElementById('short-break-duration');
const longBreakDurationInput = document.getElementById('long-break-duration');
const sessionsBeforeLongBreakInput = document.getElementById('sessions-before-long-break');
const soundEnabledInput = document.getElementById('sound-enabled');
const autoStartBreaksInput = document.getElementById('auto-start-breaks');
const autoStartWorkInput = document.getElementById('auto-start-work');
const notificationsEnabledInput = document.getElementById('notifications-enabled');

// 应用状态
let timerState = {
  minutes: 25,
  seconds: 0,
  mode: 'work', // 'work', 'shortBreak', 'longBreak'
  isRunning: false,
  timerId: null,
  sessionsCompleted: 0
};

// 用户设置（默认值）
let userSettings = {
  workDuration: 25,
  shortBreakDuration: 5,
  longBreakDuration: 15,
  sessionsBeforeLongBreak: 4,
  soundEnabled: true,
  autoStartBreaks: true,
  autoStartWork: false,
  notificationsEnabled: true
};

// 从本地存储加载设置
function loadSettings() {
  const savedSettings = localStorage.getItem('focusTimerSettings');
  if (savedSettings) {
    userSettings = JSON.parse(savedSettings);
    
    // 更新设置输入框的值
    workDurationInput.value = userSettings.workDuration;
    shortBreakDurationInput.value = userSettings.shortBreakDuration;
    longBreakDurationInput.value = userSettings.longBreakDuration;
    sessionsBeforeLongBreakInput.value = userSettings.sessionsBeforeLongBreak;
    soundEnabledInput.checked = userSettings.soundEnabled;
    autoStartBreaksInput.checked = userSettings.autoStartBreaks;
    autoStartWorkInput.checked = userSettings.autoStartWork;
    notificationsEnabledInput.checked = userSettings.notificationsEnabled;
  }
  
  // 加载完成的会话数
  const savedSessionsCount = localStorage.getItem('focusTimerSessionsCompleted');
  if (savedSessionsCount !== null) {
    timerState.sessionsCompleted = parseInt(savedSessionsCount);
    sessionsCountElement.textContent = timerState.sessionsCompleted;
  }
  
  // 加载目标会话数
  const savedSessionsGoal = localStorage.getItem('focusTimerSessionsGoal');
  if (savedSessionsGoal !== null) {
    sessionsGoalElement.textContent = savedSessionsGoal;
  }
}

// 保存设置到本地存储
function saveSettings() {
  userSettings = {
    workDuration: parseInt(workDurationInput.value),
    shortBreakDuration: parseInt(shortBreakDurationInput.value),
    longBreakDuration: parseInt(longBreakDurationInput.value),
    sessionsBeforeLongBreak: parseInt(sessionsBeforeLongBreakInput.value),
    soundEnabled: soundEnabledInput.checked,
    autoStartBreaks: autoStartBreaksInput.checked,
    autoStartWork: autoStartWorkInput.checked,
    notificationsEnabled: notificationsEnabledInput.checked
  };
  
  localStorage.setItem('focusTimerSettings', JSON.stringify(userSettings));
  resetTimer();
  
  // 隐藏设置面板
  settingsContainer.style.display = 'none';
  
  // 提供反馈
  alert('设置已保存！');
}

// 计时器逻辑
function startTimer() {
  if (timerState.isRunning) return;
  
  timerState.isRunning = true;
  startButton.disabled = true;
  pauseButton.disabled = false;
  timerElement.classList.add('timer-running');
  
  timerState.timerId = setInterval(() => {
    updateTimer();
  }, 1000);
}

function pauseTimer() {
  if (!timerState.isRunning) return;
  
  timerState.isRunning = false;
  startButton.disabled = false;
  pauseButton.disabled = true;
  timerElement.classList.remove('timer-running');
  
  clearInterval(timerState.timerId);
}

function resetTimer() {
  pauseTimer();
  
  // 根据当前模式设置时间
  if (timerState.mode === 'work') {
    timerState.minutes = userSettings.workDuration;
  } else if (timerState.mode === 'shortBreak') {
    timerState.minutes = userSettings.shortBreakDuration;
  } else {
    timerState.minutes = userSettings.longBreakDuration;
  }
  
  timerState.seconds = 0;
  updateTimerDisplay();
}

function updateTimer() {
  // 播放滴答声
  if (userSettings.soundEnabled) {
    tickSound.play().catch(err => console.error('无法播放滴答声:', err));
  }
  
  // 更新计时器
  if (timerState.seconds === 0) {
    if (timerState.minutes === 0) {
      // 计时结束
      timerComplete();
      return;
    }
    timerState.minutes--;
    timerState.seconds = 59;
  } else {
    timerState.seconds--;
  }
  
  updateTimerDisplay();
}

function updateTimerDisplay() {
  minutesElement.textContent = String(timerState.minutes).padStart(2, '0');
  secondsElement.textContent = String(timerState.seconds).padStart(2, '0');
}

function timerComplete() {
  pauseTimer();
  
  // 播放提示音
  if (userSettings.soundEnabled) {
    bellSound.play().catch(err => console.error('无法播放提示音:', err));
  }
  
  // 发送通知
  if (userSettings.notificationsEnabled) {
    showNotification();
  }
  
  // 更新会话计数（仅在工作模式结束时）
  if (timerState.mode === 'work') {
    timerState.sessionsCompleted++;
    sessionsCountElement.textContent = timerState.sessionsCompleted;
    localStorage.setItem('focusTimerSessionsCompleted', timerState.sessionsCompleted);
    
    // 决定下一个休息是短休息还是长休息
    if (timerState.sessionsCompleted % userSettings.sessionsBeforeLongBreak === 0) {
      switchMode('longBreak');
    } else {
      switchMode('shortBreak');
    }
    
    // 如果设置了自动开始休息
    if (userSettings.autoStartBreaks) {
      startTimer();
    }
  } else {
    // 从休息模式切换回工作模式
    switchMode('work');
    
    // 如果设置了自动开始工作
    if (userSettings.autoStartWork) {
      startTimer();
    }
  }
}

function switchMode(mode) {
  timerState.mode = mode;
  
  // 更新UI
  document.body.classList.remove('work-mode', 'break-mode');
  
  if (mode === 'work') {
    modeIndicatorElement.textContent = '工作时间';
    document.body.classList.add('work-mode');
    timerState.minutes = userSettings.workDuration;
  } else if (mode === 'shortBreak') {
    modeIndicatorElement.textContent = '短休息';
    document.body.classList.add('break-mode');
    timerState.minutes = userSettings.shortBreakDuration;
  } else {
    modeIndicatorElement.textContent = '长休息';
    document.body.classList.add('break-mode');
    timerState.minutes = userSettings.longBreakDuration;
  }
  
  timerState.seconds = 0;
  updateTimerDisplay();
}

// 通知功能
function showNotification() {
  // 检查浏览器是否支持通知
  if (!("Notification" in window)) {
    console.log("此浏览器不支持通知");
    return;
  }
  
  // 检查通知权限
  if (Notification.permission === "granted") {
    createNotification();
  } else if (Notification.permission !== "denied") {
    Notification.requestPermission().then(permission => {
      if (permission === "granted") {
        createNotification();
      }
    });
  }
}

function createNotification() {
  const title = timerState.mode === 'work' ? '工作时间结束!' : '休息时间结束!';
  const options = {
    body: timerState.mode === 'work' ? '休息一下吧!' : '回到工作状态!',
    icon: 'assets/favicon.ico'
  };
  
  const notification = new Notification(title, options);
  
  // 点击通知时聚焦应用窗口
  notification.onclick = function() {
    window.focus();
    notification.close();
  };
  
  // 5秒后自动关闭通知
  setTimeout(() => notification.close(), 5000);
}

// 初始化应用
function initApp() {
  // 加载设置
  loadSettings();
  
  // 初始化计时器显示
  resetTimer();
  
  // 添加事件监听器
  startButton.addEventListener('click', startTimer);
  pauseButton.addEventListener('click', pauseTimer);
  resetButton.addEventListener('click', resetTimer);
  
  // 设置面板切换
  settingsToggleButton.addEventListener('click', () => {
    settingsContainer.style.display = settingsContainer.style.display === 'none' || !settingsContainer.style.display ? 'block' : 'none';
  });
  
  // 保存设置
  saveSettingsButton.addEventListener('click', saveSettings);
  
  // 会话目标点击编辑
  sessionsGoalElement.addEventListener('click', function() {
    const newGoal = prompt('设置新的目标会话数:', this.textContent);
    if (newGoal !== null && !isNaN(newGoal) && parseInt(newGoal) > 0) {
      this.textContent = parseInt(newGoal);
      localStorage.setItem('focusTimerSessionsGoal', newGoal);
    }
  });
  
  // 请求通知权限
  if (Notification.permission !== 'granted' && Notification.permission !== 'denied') {
    Notification.requestPermission();
  }
  
  // 设置初始模式
  document.body.classList.add('work-mode');
}

// 页面加载时初始化应用
document.addEventListener('DOMContentLoaded', initApp);
```

这段JavaScript代码实现了专注时钟的所有核心功能：
- 计时器逻辑和显示
- 在工作和休息模式之间切换
- 保存和加载用户设置
- 音频提醒
- 桌面通知
- 会话计数统计

const darkModeToggle = document.getElementById('dark-mode-toggle');

// 检查是否已有保存的模式偏好
const savedTheme = localStorage.getItem('theme');
if (savedTheme === 'dark') {
  document.documentElement.classList.add('dark-theme');
  darkModeToggle.checked = true;
}

// 切换深色/浅色模式
darkModeToggle.addEventListener('change', function() {
  if (this.checked) {
    document.documentElement.classList.add('dark-theme');
    localStorage.setItem('theme', 'dark');
  } else {
    document.documentElement.classList.remove('dark-theme');
    localStorage.setItem('theme', 'light');
  }
});

## 常见问题解答

**Q: 通知功能在某些浏览器中不工作怎么办？**  
A: 不同浏览器对通知API的支持可能有所不同。确保你已经请求了通知权限，并检查浏览器的兼容性。

**Q: 如何让计时器在后台运行时也能正常工作？**  
A: 标准的setTimeout/setInterval在标签页不活跃时可能会减慢或暂停。可以考虑使用Web Workers或者计算时间差来保证计时器的准确性。

**Q: 为什么音频有时不播放？**  
A: 一些浏览器限制自动播放音频。确保在用户交互后（如点击按钮）再播放音频，或者提供静音选项。

**Q: 如何支持更多语言？**  
A: 考虑添加一个语言选择器，并为每种支持的语言创建翻译文件。使用JavaScript根据用户选择切换界面文本。
